create table if not exists cap_citizen(
	id int(6) unsigned not null auto_increment,
	fname varchar(20) not null,
	mname varchar(20) not null,
	lname varchar(20) not null,
	bdate DATE not null,
	cap_civilstatus_id int(2) unsigned not null,
	cap_disability_id int(2) unsigned not null,
	cap_citizenship_id int(2) unsigned not null,
	cap_ethnicity_id int(2) unsigned not null,
	cap_sex_id int(2) unsigned not null,
	cap_occupation_id int(2) unsigned not null,
	citID varchar(14) not null,
	primary key(id),
	key cap_civilstatus_id(cap_civilstatus_id),
	key cap_disability_id(cap_disability_id),
	key cap_citizenship_id(cap_citizenship_id),
	key cap_ethnicity_id(cap_ethnicity_id),
	key cap_sex_id(cap_sex_id),
	key cap_occupation_id(cap_occupation_id),
	constraint cap_citizen_cap_civilstatus_id foreign key(cap_civilstatus_id) references cap_civilstatus(id),
	constraint cap_citizen_cap_disability_id foreign key(cap_disability_id) references cap_disability(id),
	constraint cap_citizen_cap_citizenship_id foreign key(cap_citizenship_id) references cap_citizenship(id),
	constraint cap_citizen_cap_ethnicity_id foreign key(cap_ethnicity_id) references cap_ethnicity(id),
	constraint cap_citizen_cap_sex_id foreign key(cap_sex_id) references cap_sex(id),
	constraint cap_citizen_cap_occupation_id foreign key(cap_occupation_id) references cap_occupation(id)	
);
