create table if not exists cap_lguprofile(
	id int (5) unsigned not null auto_increment,
		cap_lguname_id int (5) unsigned not null,
		cap_lgucategory_id int (2) unsigned not null,
		cap_lgudistrict_id int (1) unsigned not null,
		cap_religion_id int (3) unsigned not null,
		cap_ethnicity_id int (3) unsigned not null,
		cap_citizenship_id int (3) unsigned not null,
		cap_lgulevel_id int (1) unsigned not null,
		name varchar (50) not null,
		website varchar (50) unique not null,
		totalbrgy int (2) unsigned not null,
		landarea varchar (50) not null,
		zipcode varchar (4) not null,
		primary key(ID),
		key cap_lguname_id(cap_lguname_id),
		key cap_lgucategory_id(cap_lgucategory_id),
		Key cap_lgudistrict_id(cap_lgudistrict_id),
		key cap_religion_id(cap_religion_id),
		key cap_ethnicity_id(cap_ethnicity_id),
		key cap_citizenship_id(cap_ethnicity_id),
		key cap_lgulevel_id(cap_lgulevel_id),
		constraint cap_lguprofile_cap_lguname_id foreign key(cap_lguname_id) references cap_lguname(id),
		constraint cap_lguprofile_cap_lgucategory_id foreign key(cap_lgucategory_id) references cap_lgucategory(id),
		constraint cap_lguprofile_cap_lgudistrict_id foreign key(cap_lgudistrict_id) references cap_lgudistrict(id),
		constraint cap_lguprofile_cap_religion_id foreign key(cap_religion_id) references cap_religion(id),
		constraint cap_lguprofile_cap_ethnicity_id foreign key(cap_ethnicity_id) references cap_ethnicity(id),
		constraint cap_lguprofile_cap_citizenship_id foreign key(cap_citizenship_id) references cap_citizenship(id),
		constraint cap_lguprofile_cap_lgulevel_id foreign key(cap_lgulevel_id) references cap_lgulevel(id)
);
